Para cada uma das bases, foram construdas diferentes ELMs (pelo menos 3 para cada base de dados) variando-se o nmero de neurnios da camada intermediria.

Neste primeiro momento, a avaliao da qualidade dos resultados ser exclusivamente visual.

1 Construindo um classificador ELM do zero¶

Foi construda a classe "ELM" que implementa o classificador ELM, conforme o cdigo a seguir:

from sklearn.base import BaseEstimator
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

class ELM(BaseEstimator):
    """ Classificador ELM """

    def __init__(self):
        self.Z = None
        self.W = None
        self.p = None
    
    def _check_X_y(self, X, y):
        """ Validate assumptions about format of input data"""
        assert set(y) == {-1, 1}, 'Response variable must be 1'
        return X, y

    def fit(self, X: np.ndarray, y: np.ndarray, p):
        """Controi um classificador otimizado a partir do conjunto de treinamento (X, y).

         Parmetros
         ----------
         X: {tipo matriz, matriz esparsa} de forma (n_amostras, n_caractersticas)
             Os exemplos de entrada de treinamento. 

         y: Vetor de formato (n_samples,)
             Os valores alvo (rtulos de classe) do conjunto de treinamento.
         
         p: nmero de neurnios da camada intermediria

         Retorna
         -------
         self: objeto
             Estimador ajustado.
         """

        X, y = self._check_X_y(X, y)
        self.p = p
        # Vetor pesos da camada escondida gerado de forma aleatria 
        self.Z = np.random.uniform(low=-0.5, high=0.5, size=(X.shape[1] +1, self.p))

        H = np.tanh(np.dot(X, self.Z[1:]) + self.Z[0])

        # Vetor pesos entre a camada de sada e a camada intermediria
        self.W = np.dot(np.linalg.pinv(H), y)

        return self

    def predict(self, X):
        """ Make predictions using already fitted model

        Parmetros
         ----------
        X: {tipo matriz, matriz esparsa} de forma (n_amostras, n_caractersticas)
             Os exemplos de entrada. 
        """

        H = np.tanh(np.dot(X, self.Z[1:]) + self.Z[0])
        return np.sign(np.dot(H,self.W))
    
    def score(self, X, Y):
        """ Retorna a acurcia do classificador

        Parmetros
         ----------
        X: {tipo matriz, matriz esparsa} de forma (n_amostras, n_caractersticas)
             Os exemplos de entrada. 

        Y: Vetor de formato (n_samples,)
             Os valores alvo (rtulos de classe) dos exemplos de entrada.
        """
        Y_pred = self.predict(X)
        return np.sum(Y_pred == Y)/Y.shape[0]

    def plot_separacao(self, X, X_1, X_2, ax):
        x_1 = np.linspace(np.floor(X[:,0].min()), np.ceil(X[:,0].max()), 200)
        x_2 = np.linspace(np.floor(X[:,1].min()), np.ceil(X[:,1].max()), 200)

        Contour = np.zeros((x_1.shape[0], x_2.shape[0]))

        for i in range(x_2.shape[0]):
            for j in range(x_1.shape[0]):
                Contour[j,i] = self.predict(np.array([x_2[i], x_1[j]]))

        
        ax.plot(X_1[:,0], X_1[:,1], 'o')
        ax.plot(X_2[:,0], X_2[:,1], 'o')
        ax.set(xlabel='X2', ylabel='X1')
        ax.pcolor(x_2, x_1, Contour, cmap="RdGy")
        # ax.set_xlim(x_1.min(), x_1.max())
        # ax.set_ylim(x_2.min(), x_2.max())
        ax.set_title('ELM: p = ' + str(self.p))

# Importao do conjunto de dados
data = pd.read_csv('./2dnormals.csv', delimiter=",", header=None)

#remoo da primeira linha
data = data[1:-1]

# Estrao dos atributos
X = data[[1,2]].values.astype(np.float)

# Extrao dos rtulos
Y = np.squeeze(data[[3]].values.astype(np.int))

# Mapeia os rtulos binrios de forma que negativo = -1 e positivo = +1
Y[Y==2] = -1

# Separa os pontos da classe positiva dos pontos da classe negativa
X_1 = np.vstack((X[Y==-1,0], X[Y==-1,1])).T
X_2 = np.vstack((X[Y==1,0], X[Y==1,1])).T

# plota os pontos
plt.scatter(X_1[:,0], X_1[:,1])
plt.scatter(X_2[:,0], X_2[:,1])
plt.title('2dnormals')
plt.xlabel('X1')
plt.ylabel('X2')

# Treina o 3 modelo : p = 30
model3 = ELM().fit(X, Y, 30)
acuracia3 = model3.score(X, Y)
model3.plot_separacao(X, X_1, X_2, ax3)

ntido que para $p = 3$ foi o obtido o separador com melhor generalizao, sendo mais robusto a varincias do dado. Para $p =30$ e $p=10$ verifica-se que ocorreu sobreajuste.
